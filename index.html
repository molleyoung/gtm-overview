columns:
  email:   "Email"
  company: "Company"
  title:   "Job Title"

pipeline:
  - cleanup          # normalize, drop junk rows
  - enrich           # backfill missing firmographics
  - sf_match         # email match -> leads + contacts; fuzzy account match
  - exclusions       # dedupe vs existing records
  - create_accounts  # net-new, territory-aware
  - create_contacts  # owner = routed rep
  - campaign_members # attach to campaign
